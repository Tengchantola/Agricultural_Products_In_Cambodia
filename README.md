# Clone the project using this URL:
    + git clone https://github.com/Tengchantola/Agricultural_Products_In_Cambodia.git
# Run the project frontend using: 
    + cd/frontend
    + npm install
    + npm run dev
    + if you have account you can login
    + if you don't have account you can register for login
# Run the project backend using:
    + cd/backend/drf
    # For Window using: venv\Scripts\activate
    # For macOS / Linux: source venv/bin/activate
    + pip install -r requirements.txt
    + connect your oracle db in setting.py
    + python manage.py runserver
